CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(reusability NONE)

SET(LATEX_COMPILER_FLAGS "-interaction=nonstopmode --enable-write18" CACHE STRING "Flags passed to latex.")

SET(CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmake/modules)
INCLUDE(${CMAKE_MODULE_PATH}/UseLATEX.cmake)

SET(SOURCE_INPUTS
  abstract.tex
  environment.tex
  design.tex
  implementation.tex
  conclusion.tex
  )

ADD_LATEX_DOCUMENT(${PROJECT_NAME}.tex INPUTS ${SOURCE_INPUTS} IMAGE_DIRS images DEFAULT_PDF MANGLE_TARGET_NAMES)
